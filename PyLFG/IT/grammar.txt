# Types

D = "Determiner" # il, lo, la, i, gli, le
N = "Noun" # gatto, tappeto, cane, libro, casa, albero
V = "Verb" # sedere, dormire, correre, leggere, vivere, arrampicarsi
P = "Preposition" # su, in, sotto, con, da, vicino
Adv = "Adverb" # velocemente, lentamente, attentamente, felicemente, silenziosamente
Adj = "Adjective" # grande, piccolo, rosso, blu, felice, verde
Pro = "Pronoun" # io, tu, lui, lei, esso, noi, voi, loro, me, lui, lei
Conj = "Conjunction" # e, o, ma, perchÃ©, mentre

# Production rules

# Simple sentences

S -> NP VP
S -> NP VP Adv
S -> NP VP PP
S -> NP VP PP Adv

# Noun phrases

NP -> D N
NP -> D N PP
NP -> D N Adj
NP -> D N Adj N
NP -> D N PP Adj N
NP -> D N Adj N PP
NP -> Pro
NP -> N

# Verb phrases

VP -> V NP NP PP
VP -> V NP NP PP Adv
VP -> V PP

# Prepositional phrases

PP -> P NP

# Compound sentences

S -> S Conj S

# Complex sentences

S -> S Conj NP VP PP
S -> S Conj NP VP PP Adv
