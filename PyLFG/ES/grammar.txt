# Types

D = "Determinador"  # el, un, la
N = "Nombre"  # gato, perro, libro, casa, árbol
V = "Verbo"  # sentarse, dormir, correr, leer, vivir, trepar
P = "Preposición"  # en, sobre, debajo, con, cerca
Adv = "Adverbio"  # rápidamente, lentamente, cuidadosamente, felizmente, silenciosamente
Adj = "Adjetivo"  # grande, pequeño, rojo, azul, feliz, verde
Pro = "Pronombre"  # yo, tú, él, ella, nosotros, ellos, mí, ti, él
Conj = "Conjunción"  # y, o, pero, porque, mientras

# Production rules

# Oraciones simples
S -> NP VP
S -> NP VP Adv
S -> NP VP PP
S -> NP VP PP Adv

# Frases nominales
NP -> D N
NP -> D N PP
NP -> D N Adj
NP -> D N Adj N
NP -> D N PP Adj N
NP -> D N Adj N PP
NP -> Pro
NP -> N

# Frases verbales
VP -> V
VP -> V NP
VP -> V NP Adv
VP -> V NP PP
VP -> V NP PP Adv
VP -> V NP NP
VP -> V NP NP Adv
VP -> V NP NP PP
VP -> V NP NP PP Adv
VP -> V PP

# Frases preposicionales
PP -> P NP

# Oraciones compuestas
S -> S Conj S

# Oraciones complejas
S -> S Conj VP
S -> S Conj NP VP
S -> S Conj NP VP Adv
S -> S Conj NP VP PP
S -> S Conj NP VP PP Adv

# Modales
VP -> Mod V
VP -> Mod V NP
VP -> Mod V NP Adv
VP -> Mod V NP PP
VP -> Mod V NP PP Adv
VP -> Mod V NP NP
VP -> Mod V NP NP Adv
VP -> Mod V NP NP PP
VP -> Mod V NP NP PP Adv
VP -> Mod V PP

Mod -> 'puede'
Mod -> 'podría'
Mod -> 'debe'
Mod -> 'debería'
Mod -> 'va a'
Mod -> 'irá'
