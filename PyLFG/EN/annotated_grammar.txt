// Types
D = "Determiner"
NP = "Noun Phrase"
VP = "Verb Phrase"
PP = "Prepositional Phrase"
Adv = "Adverb"
Adj = "Adjective"
Pro = "Pronoun"
Conj = "Conjunction"
Mod = "Modal"

// Production rules

// Simple sentences
S → NP VP [Adv] {
    ↑ = ↓1;
    (↑ SUBJ) = ↓1;
    (↑ PRED) = ↓2;
    (↑ MOD) = [MOD:↓3];
}

S → NP VP PP [Adv] {
    ↑ = ↓1;
    (↑ SUBJ) = ↓1;
    (↑ PRED) = [PRED:'VP PP'];
    (↑ MOD) = [MOD:↓4];
}

// Noun Phrases
NP → D [NP|Pro] {
    ↑ = [NOUN:↓2];
}
